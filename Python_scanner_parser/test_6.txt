program TestIfWhile;

var 
    a, b, c, i : int;
    res : float;

main
{
    a = 5;
    b = 10;
    c = 15;
    i = 0;

    print("=== INICIO DEL PROGRAMA ===");

    // Caso 1: if simple
    if (a < b) {
        print("a es menor que b");
    };

    // Caso 2: if con else
    if (c < b) {
        print("c es menor que b");
    } else {
        print("c es mayor o igual que b");
    };

    // Caso 3: while simple con incremento
    while (i < 3) do {
        print("Iteración:", i);
        i = i + 1;
    };

    // Caso 4: while con operación anidada
    i = 0;
    while (i != 2) do {
        res = (a + b * c) / (i + 1);
        print("Resultado parcial:", res);
        i = i + 1;
    };

    // Caso 5: if dentro de while
    i = 0;
    while (i < 5) do {
        if (i != 2) {
            print("i no es 2:", i);
        } else {
            print("Encontrado el 2");
        };
        i = i + 1;
    };

    print("=== FIN DEL PROGRAMA ===");
}
end
