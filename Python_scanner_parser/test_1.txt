program Fibonacci;

var
    n : int;
    resultado : int;

int fib(num : int)
var
    a, b : int;
{
    if (num < 2) {
        return(num);
    } else {
        a = fib(num - 1);
        b = fib(num - 2);
        return(a + b);
    };
};

void imprimirValor(valor : int)
{
    print(valor);
};

main
{
    n = 12;
    resultado = fib(n);
    imprimirValor(resultado);
}
end
